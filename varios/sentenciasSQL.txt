create table sobremesa (
	cod_pc_sobremesa int not null AUTO_INCREMENT PRIMARY KEY,
	nombre varchar (100) not null,
	pvp double (4,2) not null,
    	descripcion text,
   	marca varchar (20),
  	tipo varchar (10),
   	disponible boolean DEFAULT true,
  	so boolean default true,
    	oferta boolean DEFAULT false,
    	valoracion int (1)
);




create table componentes (
	cod_comp int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar (100) not null,
    pvp double (3,2) not null,
    marca varchar (20),
    tipo varchar (10),
    valoracion int (1),
    descripcion text,
    oferta boolean DEFAULT false,
    disponible boolean DEFAULT true
);





create table perifericos (
	cod_periferico int AUTO_INCREMENT not null PRIMARY KEY,
    nombre varchar (100) not null,
    pvp float (3,2) not null,
    marca varchar (20),
    tipo varchar (10),
    disponible boolean default true,
    oferta boolean default false,
    descripcion text,
    valoracion int (1)
);




create table movil (
	cod_movil int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar (100) not null,
    pvp double (3,2) not null,
    marca varchar (20) not null,
    tipo varchar (10) not null,
    disponible boolean DEFAULT true,
    oferta boolean DEFAULT false,
    descripcion text,
    valoracion int (1)
);




create table clientes (
	cod_cliente int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar (40) not null,
    apellidos varchar (50) not null,
    usuario varchar (15) not null,
    pass varchar (15) not null,
    dni varchar (9) not null UNIQUE,
    fecha_nacimiento date,
    correo varchar (50),
    pais varchar (20) not null,
    provincia varchar (20) not null,
    localidad varchar (20) not null,
    domicilio varchar (50) not null,
    cod_postal varchar (5) not null,
    num_pedidos int (2) default 0
);




create table accesorios (
	cod_ac int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar (40),
    dispositivo varchar (20),
    material varchar (20),
    marca varchar (20),
    pulgadas varchar (10),
    descripcion text
);






create table tablets (
	cod_tablet int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar (100) not null,
    pvp double (3,2) not null,
    marca varchar (20),
    descripcion text,
    disponible boolean default true,
    oferta boolean DEFAULT false,
    valoracion int (1)
);





create table contacto (
	cod_mensaje int not null AUTO_INCREMENT PRIMARY KEY,
    nombre varchar (20),
    apellidos varchar (50),
    pais varchar (20),
    mensaje text
);




create table portatil (
	cod_pc_portatil int not null AUTO_INCREMENT PRIMARY KEY,
	nombre varchar (100) not null,
	pvp double (4,2) not null,
    	descripcion text,
   	marca varchar (20),
  	tipo varchar (10),
   	disponible boolean DEFAULT true,
  	so boolean default true,
    	oferta boolean DEFAULT false,
    	valoracion int (1)
);



----PRUEBAS------




create table inventario (
	cod_pc_sobremesa int not null,
	cod_pc_portatil int not null,
	cod_componente int not null,
	cod_periferico int not null,
	cod_movil int not null,
	cod_accesorio int not null,
	cod_tablet int not null,
	constraint cod_pc_sobremesa FOREIGN KEY references sobremesa(cod_pc_sobremesa),
	constraint cod_pc_portatil foreing key references portatil(cod_pc_portatil),
	constraint cod_componente foreing key references componentes(cod_comp),
	constraint cod_periferico foreing key references perifericos(cod_periferico),
	constraint cod_movil foreing key references movil(cod_movil),
	constraint cod_accesorio foreing key references accesorios(cod_ac),
	constraint cod_tablet foreing key cod_tablet references tablets(cod_tablet)
);

	FOREIGN KEY cod_pc_sobremesa references sobremesa(cod_pc_sobremesa),
	foreing key cod_pc_portatil references portatil(cod_pc_portatil),
	foreing key cod_componente references componentes(cod_comp),
	foreing key cod_periferico references perifericos(cod_periferico),
	foreing key cod_movil references movil(cod_movil),
	foreing key cod_accesorio references accesorios(cod_ac),
	foreing key cod_tablet references tablets(cod_tablet)
